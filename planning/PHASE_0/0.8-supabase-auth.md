## Steps

### Step 1: Configure Auth Providers in Supabase

1. Go to Supabase Dashboard → **Authentication** → **Providers**
2. Ensure **Email** is enabled:
   - Enable email confirmations: **OFF** (for MVP)
   - Secure email change: **ON**
   - Secure password change: **ON**
3. Leave other providers disabled for now

### Step 2: Configure URL Settings

1. Go to **Authentication** → **URL Configuration**
2. Set:
   - **Site URL**: `http://localhost:3000` (update to production URL later)
   - **Redirect URLs**: Add:
     - `http://localhost:3000/**`
     - `https://legacyscale-saas.vercel.app/**`
     - `https://app.legacyscale.co/**` (if using custom domain)

### Step 3: Create Auth Callback Route

Create `app/auth/callback/route.ts`:

```bash
mkdir -p app/auth/callback

cat > app/auth/callback/route.ts << 'EOF'
import { createClient } from '@/lib/supabase/server'
import { NextResponse } from 'next/server'

export async function GET(request: Request) {
  const { searchParams, origin } = new URL(request.url)
  const code = searchParams.get('code')
  const next = searchParams.get('next') ?? '/dashboard'

  if (code) {
    const supabase = await createClient()
    const { error } = await supabase.auth.exchangeCodeForSession(code)

    if (!error) {
      return NextResponse.redirect(`${origin}${next}`)
    }
  }

  // Return the user to an error page with instructions
  return NextResponse.redirect(`${origin}/auth/auth-code-error`)
}
EOF
```

### Step 5: Create User Sync Function

When a user signs up or logs in, sync their profile to our users table.

Create `lib/auth/sync-user.ts`:

```bash
mkdir -p lib/auth

cat > lib/auth/sync-user.ts << 'EOF'
import { prisma } from '@/lib/prisma'
import { User } from '@supabase/supabase-js'

export async function syncUserToDatabase(authUser: User) {
  if (!authUser.email) {
    throw new Error('User email is required')
  }

  // Upsert user in our database
  const user = await prisma.user.upsert({
    where: { id: authUser.id },
    update: {
      email: authUser.email,
      name: authUser.user_metadata?.name || authUser.email.split('@')[0],
      avatarUrl: authUser.user_metadata?.avatar_url,
    },
    create: {
      id: authUser.id,
      email: authUser.email,
      name: authUser.user_metadata?.name || authUser.email.split('@')[0],
      avatarUrl: authUser.user_metadata?.avatar_url,
      role: 'MEMBER', // Default role
      isActive: true,
    },
  })

  return user
}
EOF
```

### Step 6: Create Auth Actions

Create `lib/auth/actions.ts`:

```bash
cat > lib/auth/actions.ts << 'EOF'
'use server'

import { createClient } from '@/lib/supabase/server'
import { redirect } from 'next/navigation'
import { syncUserToDatabase } from './sync-user'

export async function signIn(formData: FormData) {
  const email = formData.get('email') as string
  const password = formData.get('password') as string

  const supabase = await createClient()

  const { data, error } = await supabase.auth.signInWithPassword({
    email,
    password,
  })

  if (error) {
    return { error: error.message }
  }

  if (data.user) {
    // Sync user to database
    await syncUserToDatabase(data.user)
  }

  redirect('/dashboard')
}

export async function signUp(formData: FormData) {
  const email = formData.get('email') as string
  const password = formData.get('password') as string
  const name = formData.get('name') as string

  const supabase = await createClient()

  const { data, error } = await supabase.auth.signUp({
    email,
    password,
    options: {
      data: {
        name,
      },
    },
  })

  if (error) {
    return { error: error.message }
  }

  if (data.user) {
    // Sync user to database
    await syncUserToDatabase(data.user)
  }

  redirect('/dashboard')
}

export async function signOut() {
  const supabase = await createClient()
  await supabase.auth.signOut()
  redirect('/login')
}

export async function resetPassword(formData: FormData) {
  const email = formData.get('email') as string

  const supabase = await createClient()

  const { error } = await supabase.auth.resetPasswordForEmail(email, {
    redirectTo: `${process.env.NEXT_PUBLIC_APP_URL}/reset-password`,
  })

  if (error) {
    return { error: error.message }
  }

  return { success: true }
}

export async function updatePassword(formData: FormData) {
  const password = formData.get('password') as string

  const supabase = await createClient()

  const { error } = await supabase.auth.updateUser({
    password,
  })

  if (error) {
    return { error: error.message }
  }

  redirect('/dashboard')
}
EOF
```

### Step 7: Create Get User Helper

Create `lib/auth/get-user.ts`:

```bash
cat > lib/auth/get-user.ts << 'EOF'
import { createClient } from '@/lib/supabase/server'
import { prisma } from '@/lib/prisma'
import { redirect } from 'next/navigation'

export async function getUser() {
  const supabase = await createClient()

  const { data: { user: authUser } } = await supabase.auth.getUser()

  if (!authUser) {
    return null
  }

  // Get user from our database with role info
  const user = await prisma.user.findUnique({
    where: { id: authUser.id },
  })

  return user
}

export async function requireUser() {
  const user = await getUser()

  if (!user) {
    redirect('/login')
  }

  return user
}

export async function requireAdmin() {
  const user = await requireUser()

  if (user.role !== 'ADMIN') {
    redirect('/dashboard')
  }

  return user
}

export async function requireActiveUser() {
  const user = await requireUser()

  if (!user.isActive) {
    redirect('/access-revoked')
  }

  return user
}
EOF
```

## Auth Flow Summary

### Sign Up Flow
1. User submits email + password
2. Supabase creates auth user
3. `syncUserToDatabase` creates row in users table
4. User redirected to dashboard

### Sign In Flow
1. User submits email + password
2. Supabase validates credentials
3. `syncUserToDatabase` updates user row
4. User redirected to dashboard

### Password Reset Flow
1. User submits email on forgot password page
2. Supabase sends reset email
3. User clicks link, lands on reset password page
4. User submits new password
5. Supabase updates password
6. User redirected to dashboard

## Troubleshooting

### "Email not confirmed"
- Disable email confirmation in Supabase Auth settings
- Or implement email confirmation flow

### User not in database after signup
- Check `syncUserToDatabase` is being called
- Check Prisma connection is working
- Check user ID format matches (UUID)

### Redirect loops
- Check redirect URLs in Supabase match your app URLs
- Check middleware is not blocking auth routes

### "Invalid login credentials"
- Check email and password are correct
- Check user exists in Supabase Auth
