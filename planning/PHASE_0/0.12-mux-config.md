### Step 5: Create Mux Client Utility

Create `lib/mux.ts`:

```bash
cat > lib/mux.ts << 'EOF'
import Mux from '@mux/mux-node'

if (!process.env.MUX_TOKEN_ID || !process.env.MUX_TOKEN_SECRET) {
  throw new Error('Missing Mux API credentials')
}

// Initialize Mux client
const mux = new Mux({
  tokenId: process.env.MUX_TOKEN_ID,
  tokenSecret: process.env.MUX_TOKEN_SECRET,
})

export const { video: muxVideo, data: muxData } = mux

// ============================================
// UPLOAD FUNCTIONS
// ============================================

interface CreateUploadOptions {
  corsOrigin?: string
  newAssetSettings?: {
    playbackPolicy?: Array<'public' | 'signed'>
    passthrough?: string
  }
}

/**
 * Create a direct upload URL for client-side uploads
 */
export async function createDirectUpload(options: CreateUploadOptions = {}) {
  const { corsOrigin = '*', newAssetSettings = {} } = options

  const upload = await muxVideo.uploads.create({
    cors_origin: corsOrigin,
    new_asset_settings: {
      playback_policy: newAssetSettings.playbackPolicy || ['public'],
      passthrough: newAssetSettings.passthrough,
    },
  })

  return {
    uploadId: upload.id,
    uploadUrl: upload.url,
  }
}

/**
 * Create an asset from a URL (server-side upload)
 */
export async function createAssetFromUrl(
  url: string,
  passthrough?: string
) {
  const asset = await muxVideo.assets.create({
    input: [{ url }],
    playback_policy: ['public'],
    passthrough,
  })

  return {
    assetId: asset.id,
    playbackId: asset.playback_ids?.[0]?.id,
    status: asset.status,
  }
}

// ============================================
// ASSET FUNCTIONS
// ============================================

/**
 * Get asset details
 */
export async function getAsset(assetId: string) {
  const asset = await muxVideo.assets.retrieve(assetId)

  return {
    id: asset.id,
    status: asset.status,
    playbackId: asset.playback_ids?.[0]?.id,
    duration: asset.duration,
    aspectRatio: asset.aspect_ratio,
    resolution: asset.resolution_tier,
    createdAt: asset.created_at,
  }
}

/**
 * Delete an asset
 */
export async function deleteAsset(assetId: string) {
  await muxVideo.assets.delete(assetId)
  return { success: true }
}

/**
 * Get upload status
 */
export async function getUploadStatus(uploadId: string) {
  const upload = await muxVideo.uploads.retrieve(uploadId)

  return {
    id: upload.id,
    status: upload.status,
    assetId: upload.asset_id,
  }
}

// ============================================
// PLAYBACK FUNCTIONS
// ============================================

/**
 * Get playback URL for a video
 */
export function getPlaybackUrl(playbackId: string) {
  return `https://stream.mux.com/${playbackId}.m3u8`
}

/**
 * Get thumbnail URL for a video
 */
export function getThumbnailUrl(
  playbackId: string,
  options: {
    width?: number
    height?: number
    time?: number
    fitMode?: 'preserve' | 'stretch' | 'crop' | 'smartcrop' | 'pad'
  } = {}
) {
  const { width = 640, height = 360, time = 0, fitMode = 'smartcrop' } = options

  const params = new URLSearchParams({
    width: width.toString(),
    height: height.toString(),
    time: time.toString(),
    fit_mode: fitMode,
  })

  return `https://image.mux.com/${playbackId}/thumbnail.jpg?${params}`
}

/**
 * Get animated GIF preview URL
 */
export function getGifUrl(
  playbackId: string,
  options: {
    width?: number
    fps?: number
    start?: number
    end?: number
  } = {}
) {
  const { width = 320, fps = 15, start = 0, end = 5 } = options

  const params = new URLSearchParams({
    width: width.toString(),
    fps: fps.toString(),
    start: start.toString(),
    end: end.toString(),
  })

  return `https://image.mux.com/${playbackId}/animated.gif?${params}`
}

/**
 * Get storyboard sprite URL for video scrubbing
 */
export function getStoryboardUrl(playbackId: string) {
  return `https://image.mux.com/${playbackId}/storyboard.vtt`
}

// ============================================
// WEBHOOK HELPERS
// ============================================

/**
 * Verify Mux webhook signature
 */
export function verifyWebhookSignature(
  rawBody: string,
  signature: string,
  secret: string
): boolean {
  const crypto = require('crypto')
  const expectedSignature = crypto
    .createHmac('sha256', secret)
    .update(rawBody)
    .digest('hex')

  return signature === expectedSignature
}

// Export types
export type MuxAssetStatus = 'preparing' | 'ready' | 'errored'
export type MuxUploadStatus = 'waiting' | 'asset_created' | 'errored' | 'cancelled' | 'timed_out'
```

### Step 6: Create Mux Webhook Handler

Create `app/api/webhooks/mux/route.ts`:

```bash
mkdir -p app/api/webhooks/mux

cat > app/api/webhooks/mux/route.ts << 'EOF'
import { NextRequest, NextResponse } from 'next/server'
import { headers } from 'next/headers'
import { prisma } from '@/lib/prisma'
import { verifyWebhookSignature } from '@/lib/mux'

export async function POST(request: NextRequest) {
  try {
    const rawBody = await request.text()
    const headersList = await headers()
    const signature = headersList.get('mux-signature')

    // Verify webhook signature in production
    if (process.env.NODE_ENV === 'production') {
      if (!signature || !process.env.MUX_WEBHOOK_SECRET) {
        return NextResponse.json(
          { error: 'Missing signature or webhook secret' },
          { status: 401 }
        )
      }

      const isValid = verifyWebhookSignature(
        rawBody,
        signature,
        process.env.MUX_WEBHOOK_SECRET
      )

      if (!isValid) {
        return NextResponse.json(
          { error: 'Invalid signature' },
          { status: 401 }
        )
      }
    }

    const event = JSON.parse(rawBody)

    console.log('Mux webhook received:', event.type)

    switch (event.type) {
      case 'video.asset.ready': {
        // Video finished processing
        const { id: assetId, playback_ids, duration, passthrough } = event.data

        const playbackId = playback_ids?.[0]?.id
        const durationSeconds = Math.round(duration || 0)

        // If passthrough contains lessonId, update the lesson
        if (passthrough) {
          await prisma.lesson.update({
            where: { id: passthrough },
            data: {
              muxAssetId: assetId,
              muxPlaybackId: playbackId,
              durationSeconds,
              status: 'READY',
            },
          })
          console.log(`Lesson ${passthrough} updated with Mux asset`)
        }
        break
      }

      case 'video.asset.errored': {
        // Video processing failed
        const { passthrough } = event.data

        if (passthrough) {
          await prisma.lesson.update({
            where: { id: passthrough },
            data: {
              status: 'DRAFT', // Reset to draft so user can retry
            },
          })
          console.error(`Lesson ${passthrough} video processing failed`)
        }
        break
      }

      case 'video.asset.deleted': {
        // Video was deleted
        const { id: assetId } = event.data

        // Find and update any lessons with this asset
        await prisma.lesson.updateMany({
          where: { muxAssetId: assetId },
          data: {
            muxAssetId: null,
            muxPlaybackId: null,
            durationSeconds: null,
            status: 'DRAFT',
          },
        })
        console.log(`Lessons with asset ${assetId} cleared`)
        break
      }

      case 'video.upload.asset_created': {
        // Upload completed, asset is being processed
        const { asset_id, passthrough } = event.data

        if (passthrough) {
          await prisma.lesson.update({
            where: { id: passthrough },
            data: {
              muxAssetId: asset_id,
              status: 'PROCESSING',
            },
          })
          console.log(`Lesson ${passthrough} upload completed, processing...`)
        }
        break
      }

      default:
        console.log(`Unhandled Mux event type: ${event.type}`)
    }

    return NextResponse.json({ received: true })
  } catch (error) {
    console.error('Mux webhook error:', error)
    return NextResponse.json(
      { error: 'Webhook processing failed' },
      { status: 500 }
    )
  }
}
EOF
```

```bash
import { NextRequest, NextResponse } from 'next/server'
import { requireAdmin } from '@/lib/auth'
import { createDirectUpload } from '@/lib/mux'

export async function POST(request: NextRequest) {
  try {
    // Only admins can upload videos
    await requireAdmin()

    const body = await request.json()
    const { lessonId } = body

    if (!lessonId) {
      return NextResponse.json(
        { error: 'lessonId is required' },
        { status: 400 }
      )
    }

    // Create direct upload with lesson ID as passthrough
    const upload = await createDirectUpload({
      corsOrigin: process.env.NEXT_PUBLIC_APP_URL || '*',
      newAssetSettings: {
        playbackPolicy: ['public'],
        passthrough: lessonId, // Will be returned in webhook
      },
    })

    return NextResponse.json({
      uploadId: upload.uploadId,
      uploadUrl: upload.uploadUrl,
    })
  } catch (error) {
    console.error('Video upload creation error:', error)

    if (error instanceof Error && error.message.includes('redirect')) {
      // User not authenticated
      return NextResponse.json(
        { error: 'Unauthorized' },
        { status: 401 }
      )
    }

    return NextResponse.json(
      { error: 'Failed to create upload' },
      { status: 500 }
    )
  }
}
EOF
```

### Step 8: Create Video Player Component

Create `components/video/mux-player.tsx`:

```bash
mkdir -p components/video

cat > components/video/mux-player.tsx << 'EOF'
'use client'

import { useEffect, useRef } from 'react'
import { cn } from '@/lib/utils'

interface MuxPlayerProps {
  playbackId: string
  title?: string
  poster?: string
  autoPlay?: boolean
  muted?: boolean
  loop?: boolean
  className?: string
  onTimeUpdate?: (currentTime: number) => void
  onEnded?: () => void
  startTime?: number
}

export function MuxPlayer({
  playbackId,
  title,
  poster,
  autoPlay = false,
  muted = false,
  loop = false,
  className,
  onTimeUpdate,
  onEnded,
  startTime = 0,
}: MuxPlayerProps) {
  const playerRef = useRef<HTMLVideoElement>(null)

  useEffect(() => {
    // Load hls.js for HLS playback support
    const loadHls = async () => {
      if (!playerRef.current) return

      const video = playerRef.current
      const src = `https://stream.mux.com/${playbackId}.m3u8`

      // Check if browser supports HLS natively (Safari)
      if (video.canPlayType('application/vnd.apple.mpegurl')) {
        video.src = src
      } else {
        // Use hls.js for other browsers
        const Hls = (await import('hls.js')).default

        if (Hls.isSupported()) {
          const hls = new Hls()
          hls.loadSource(src)
          hls.attachMedia(video)
        }
      }

      // Set start time if provided
      if (startTime > 0) {
        video.currentTime = startTime
      }
    }

    loadHls()
  }, [playbackId, startTime])

  const handleTimeUpdate = () => {
    if (playerRef.current && onTimeUpdate) {
      onTimeUpdate(playerRef.current.currentTime)
    }
  }

  const handleEnded = () => {
    if (onEnded) {
      onEnded()
    }
  }

  const thumbnailUrl = poster || `https://image.mux.com/${playbackId}/thumbnail.jpg?width=1280&height=720&fit_mode=smartcrop`

  return (
    <div className={cn('video-container', className)}>
      <video
        ref={playerRef}
        className="h-full w-full"
        controls
        playsInline
        autoPlay={autoPlay}
        muted={muted}
        loop={loop}
        poster={thumbnailUrl}
        onTimeUpdate={handleTimeUpdate}
        onEnded={handleEnded}
        title={title}
      >
        <track kind="captions" />
      </video>
    </div>
  )
}
EOF
```

### Step 10: Create Video Upload Component

Create `components/video/video-upload.tsx`:

```bash
cat > components/video/video-upload.tsx << 'EOF'
'use client'

import { useState, useRef } from 'react'
import { Upload, X, CheckCircle, AlertCircle, Loader2 } from 'lucide-react'
import { Button } from '@/components/ui/button'
import { Progress } from '@/components/ui/progress'
import { cn } from '@/lib/utils'

interface VideoUploadProps {
  lessonId: string
  onUploadComplete?: () => void
  className?: string
}

type UploadStatus = 'idle' | 'preparing' | 'uploading' | 'processing' | 'complete' | 'error'

export function VideoUpload({
  lessonId,
  onUploadComplete,
  className,
}: VideoUploadProps) {
  const [status, setStatus] = useState<UploadStatus>('idle')
  const [progress, setProgress] = useState(0)
  const [error, setError] = useState<string | null>(null)
  const inputRef = useRef<HTMLInputElement>(null)

  const handleFileSelect = async (event: React.ChangeEvent<HTMLInputElement>) => {
    const file = event.target.files?.[0]
    if (!file) return

    // Validate file type
    if (!file.type.startsWith('video/')) {
      setError('Please select a video file')
      return
    }

    // Validate file size (max 5GB)
    const maxSize = 5 * 1024 * 1024 * 1024
    if (file.size > maxSize) {
      setError('File size must be less than 5GB')
      return
    }

    setError(null)
    setStatus('preparing')

    try {
      // Get upload URL from our API
      const response = await fetch('/api/uploads/video', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ lessonId }),
      })

      if (!response.ok) {
        throw new Error('Failed to create upload')
      }

      const { uploadUrl } = await response.json()

      // Upload directly to Mux
      setStatus('uploading')

      const xhr = new XMLHttpRequest()

      xhr.upload.addEventListener('progress', (event) => {
        if (event.lengthComputable) {
          const percent = Math.round((event.loaded / event.total) * 100)
          setProgress(percent)
        }
      })

      xhr.addEventListener('load', () => {
        if (xhr.status === 200) {
          setStatus('processing')
          setProgress(100)

          // The webhook will update the lesson when processing is complete
          // For now, we just show processing state
          setTimeout(() => {
            setStatus('complete')
            onUploadComplete?.()
          }, 2000)
        } else {
          throw new Error('Upload failed')
        }
      })

      xhr.addEventListener('error', () => {
        throw new Error('Upload failed')
      })

      xhr.open('PUT', uploadUrl)
      xhr.send(file)
    } catch (err) {
      console.error('Upload error:', err)
      setError(err instanceof Error ? err.message : 'Upload failed')
      setStatus('error')
    }
  }

  const handleReset = () => {
    setStatus('idle')
    setProgress(0)
    setError(null)
    if (inputRef.current) {
      inputRef.current.value = ''
    }
  }

  return (
    <div className={cn('space-y-4', className)}>
      <input
        ref={inputRef}
        type="file"
        accept="video/*"
        onChange={handleFileSelect}
        className="hidden"
        disabled={status !== 'idle' && status !== 'error'}
      />

      {status === 'idle' && (
        <div
          className="flex flex-col items-center justify-center rounded-lg border-2 border-dashed p-8 text-center cursor-pointer hover:border-primary/50 transition-colors"
          onClick={() => inputRef.current?.click()}
        >
          <Upload className="h-10 w-10 text-muted-foreground mb-4" />
          <p className="text-sm font-medium">Click to upload video</p>
          <p className="text-xs text-muted-foreground mt-1">
            MP4, MOV, or WebM up to 5GB
          </p>
        </div>
      )}

      {(status === 'preparing' || status === 'uploading') && (
        <div className="rounded-lg border p-4 space-y-4">
          <div className="flex items-center gap-3">
            <Loader2 className="h-5 w-5 animate-spin text-primary" />
            <span className="text-sm font-medium">
              {status === 'preparing' ? 'Preparing upload...' : `Uploading... ${progress}%`}
            </span>
          </div>
          <Progress value={progress} />
        </div>
      )}

      {status === 'processing' && (
        <div className="rounded-lg border p-4">
          <div className="flex items-center gap-3">
            <Loader2 className="h-5 w-5 animate-spin text-primary" />
            <div>
              <p className="text-sm font-medium">Processing video...</p>
              <p className="text-xs text-muted-foreground">
                This may take a few minutes
              </p>
            </div>
          </div>
        </div>
      )}

      {status === 'complete' && (
        <div className="rounded-lg border border-success/50 bg-success/10 p-4">
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-3">
              <CheckCircle className="h-5 w-5 text-success" />
              <span className="text-sm font-medium text-success">
                Upload complete!
              </span>
            </div>
            <Button variant="ghost" size="sm" onClick={handleReset}>
              Upload another
            </Button>
          </div>
        </div>
      )}

      {status === 'error' && (
        <div className="rounded-lg border border-destructive/50 bg-destructive/10 p-4">
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-3">
              <AlertCircle className="h-5 w-5 text-destructive" />
              <div>
                <p className="text-sm font-medium text-destructive">Upload failed</p>
                <p className="text-xs text-destructive/80">{error}</p>
              </div>
            </div>
            <Button variant="ghost" size="sm" onClick={handleReset}>
              <X className="h-4 w-4" />
            </Button>
          </div>
        </div>
      )}
    </div>
  )
}
EOF
```

## Video Processing Flow

```
1. Admin clicks "Upload Video"
2. Frontend calls /api/uploads/video
3. API creates Mux direct upload URL
4. Frontend uploads video directly to Mux
5. Mux sends webhook: video.upload.asset_created
6. Webhook updates lesson status to "PROCESSING"
7. Mux processes video (encoding, thumbnails)
8. Mux sends webhook: video.asset.ready
9. Webhook updates lesson with playback ID
10. Video is ready for playback
```
