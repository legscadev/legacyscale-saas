### Step 3: Create Additional Directories

```bash
mkdir -p lib
mkdir -p components/ui
mkdir -p components/layout
mkdir -p components/shared

mkdir -p prisma

```

### Step 4: Create .env.example

```bash
cat > .env.example << 'EOF'
# Supabase
NEXT_PUBLIC_SUPABASE_URL=
NEXT_PUBLIC_SUPABASE_ANON_KEY=
SUPABASE_SERVICE_ROLE_KEY=
DATABASE_URL=

# Mux
MUX_TOKEN_ID=
MUX_TOKEN_SECRET=
MUX_WEBHOOK_SECRET=

# Resend
RESEND_API_KEY=

# GoHighLevel
GHL_WEBHOOK_SECRET=

# App
NEXT_PUBLIC_APP_URL=http://localhost:3000
EOF
```

### Step 5: Create .env.local (Copy from .env.example)

```bash
cp .env.example .env.local
```

### Step 6: Update .gitignore

Add these lines to `.gitignore`:

```bash
cat >> .gitignore << 'EOF'

# Environment files
.env.local
.env.production.local

# Prisma
prisma/migrations

# IDE
.idea/
.vscode/

# Vercel
.vercel

# Planning docs (optional)
planning/PHASE_0/*.local.md
EOF
```

### Step 7: Create lib/utils.ts

```bash
cat > lib/utils.ts << 'EOF'
import { type ClassValue, clsx } from "clsx"
import { twMerge } from "tailwind-merge"

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}
EOF
```

### Step 8: Install clsx and tailwind-merge

```bash
pnpm add clsx tailwind-merge
```

### Step 9: Update ESLint Config

Replace `.eslintrc.json`:

```bash
cat > .eslintrc.json << 'EOF'
{
  "extends": [
    "next/core-web-vitals",
    "plugin:@typescript-eslint/recommended"
  ],
  "parser": "@typescript-eslint/parser",
  "plugins": ["@typescript-eslint"],
  "rules": {
    "@typescript-eslint/no-unused-vars": "error",
    "@typescript-eslint/no-explicit-any": "error",
    "prefer-const": "error"
  },
  "ignorePatterns": ["node_modules/", ".next/", "out/"]
}
EOF
```

### Step 10: Install TypeScript ESLint

```bash
pnpm add -D @typescript-eslint/parser @typescript-eslint/eslint-plugin
```

### Step 11: Update package.json Scripts

Add these scripts to `package.json`:

```json
{
  "scripts": {
    "dev": "next dev",
    "build": "next build",
    "start": "next start",
    "lint": "next lint",
    "lint:fix": "next lint --fix",
    "type-check": "tsc --noEmit",
    "format": "prettier --write .",
    "format:check": "prettier --check ."
  }
}
```

Run:

```bash
pnpm pkg set scripts.lint:fix="next lint --fix"
pnpm pkg set scripts.type-check="tsc --noEmit"
```

### Step 12: Create types/index.ts

```bash
cat > types/index.ts << 'EOF'
// User roles
export type Role = 'ADMIN' | 'MEMBER'

// Course status
export type CourseStatus = 'DRAFT' | 'PUBLISHED' | 'ARCHIVED'

// Lesson types
export type LessonType = 'VIDEO' | 'QUIZ' | 'RESOURCE'

// Lesson status
export type LessonStatus = 'DRAFT' | 'PROCESSING' | 'READY'

// Quiz question types
export type QuestionType = 'MULTIPLE_CHOICE' | 'TRUE_FALSE'

// Announcement status
export type AnnouncementStatus = 'DRAFT' | 'PUBLISHED'
EOF
```

### Step 13: Verify Installation

```bash
# Run dev server
pnpm dev

# In another terminal, check lint
pnpm lint

# Check types
pnpm type-check
```

### Step 14: Initialize Git (if not already)

```bash
git init
git add .
git commit -m "chore: initialize Next.js 14 project with TypeScript and Tailwind"
```

## Troubleshooting

### "pnpm: command not found"
```bash
npm install -g pnpm
```

### "Node version too old"
```bash
# Use nvm to install Node 20
nvm install 20
nvm use 20
```

### ESLint errors about TypeScript
Make sure `@typescript-eslint/parser` and `@typescript-eslint/eslint-plugin` are installed:
```bash
pnpm add -D @typescript-eslint/parser @typescript-eslint/eslint-plugin
```

### Port 3000 already in use
```bash
pnpm dev -- -p 3001
```
